
[7576번: 토마토](https://www.acmicpc.net/problem/7576)

C++14 
BFS 최소값 구하기 문제
```c++
#include <cstdio>
#include <queue>
using namespace std;
int n, m;
int map[1001][1001];
int check[1001][1001];			// 익는 데 소요된 일수
int dx[4] = { -1, 1, 0, 0 };	// 상하좌우 x좌표 변동값
int dy[4] = { 0, 0, -1, 1 };	// 상하좌우 y좌표 변동값
queue<pair<int, int>> q;

// x, y가 map의 범위내의 index 값인지 검사
bool isIn(int x, int y){
	return x >= 0 && x < m && y >= 0 && y < n;
}

void bfs(){
	// queue가 빌때까지 반복
	while (!q.empty()) {
		// 탐색을 시작할 좌표
		pair<int, int> xy = q.front();
		int x = xy.first, y = xy.second;
		q.pop();

		for (int i = 0; i < 4; i++){
			// 다음으로 탐색할 좌표 후보
			int nx = x + dx[i];
			int ny = y + dy[i];

			// 토마토가 존재하는 칸일 경우
			if (map[nx][ny] == 0 && isIn(nx, ny) && check[nx][ny] == -1) {
				check[nx][ny] = check[x][y] + 1;
				q.push(make_pair(nx, ny));
			}
		}
	}
	return;
}

int main(){
	// 입력값 받기
	scanf("%d %d", &n, &m);
	for (int i = 0; i < m; i++){
		for (int j = 0; j < n; j++){
			scanf("%d", &map[i][j]);
			// check 배열 초기화
			check[i][j] = -1;
			if (map[i][j] == 1){
				// 익은 토마토가 들어있는 칸 = 시작점
				// 시작점(i, j) 넣어주기
				check[i][j] = 0;	// 처음부터 익은 토마토가 들어있는 칸은 0일 소요
				q.push(make_pair(i, j));
			}
		}
	}

	// bfs 를 통해 최소값 구하기
	bfs();
	int mindate = 0;
	for (int i = 0; i < m; i++){
		for (int j = 0; j < n; j++){
			mindate = max(mindate, check[i][j]);
			if (check[i][j] == -1 && map[i][j] != -1){
				// 결과 출력
				printf("%d\n", -1);
				return 0;
			}
		}
	}

	// 결과 출력
	printf("%d\n", mindate);
	return 0;
}
```
