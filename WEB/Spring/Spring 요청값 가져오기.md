## 매개변수를 통해 요청 값을 받아오는 방식
요청: contextPath/join?id=id1&pwd=pwd1&name=name1&addr2=addr2&buildingno=buildingno
```java
@RequestMapping("/join")
public ModelAndView join(String id
					   , String pwd
					   , String name
					   , String addr2
					   , String buildingno) {
	// Customer 객체 생성
	Customer c = new Customer();
	c.setId(id);
	c.setPwd(pwd);
	c.setName(name);
	c.setAddr(addr2);
	Post p = new Post();
	p.setBuildingno(buildingno);
	c.setPost(p);
}
	
```
